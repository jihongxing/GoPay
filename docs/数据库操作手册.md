# 数据库常用操作命令

## 启动数据库

```bash
# 启动 PostgreSQL
docker-compose up -d

# 查看日志
docker-compose logs -f postgres

# 停止数据库
docker-compose down
```

## 连接数据库

```bash
# 使用 Docker 连接
docker-compose exec postgres psql -U gopay -d gopay

# 使用本地 psql 连接
psql -h localhost -p 5432 -U gopay -d gopay
```

## 常用 SQL 命令

### 查看所有表

```sql
\dt
```

### 查看表结构

```sql
\d orders
\d+ orders  -- 详细信息
```

### 查看索引

```sql
\di
```

### 查看迁移历史

```sql
SELECT * FROM schema_migrations ORDER BY applied_at;
```

### 查询订单

```sql
-- 查询所有订单
SELECT order_no, app_id, amount, status, notify_status, created_at 
FROM orders 
ORDER BY created_at DESC 
LIMIT 10;

-- 查询待通知订单
SELECT order_no, retry_count, created_at 
FROM orders 
WHERE notify_status = 'pending' 
ORDER BY created_at;

-- 查询失败订单
SELECT order_no, status, notify_status, retry_count 
FROM orders 
WHERE notify_status = 'failed_notify';
```

### 查询通知日志

```sql
-- 查询某个订单的通知日志
SELECT 
    callback_url, 
    success, 
    response_status, 
    duration_ms, 
    created_at 
FROM notify_logs 
WHERE order_no = 'ORDER_NO_HERE' 
ORDER BY created_at DESC;

-- 统计通知成功率
SELECT 
    success,
    COUNT(*) as count,
    ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER(), 2) as percentage
FROM notify_logs 
GROUP BY success;
```

### 查询交易流水

```sql
-- 查询某个订单的所有流水
SELECT 
    transaction_no, 
    type, 
    amount, 
    status, 
    created_at 
FROM transactions 
WHERE order_no = 'ORDER_NO_HERE' 
ORDER BY created_at;
```

## 数据库维护

### 清理测试数据

```sql
-- 删除所有订单（谨慎操作！）
TRUNCATE orders CASCADE;
TRUNCATE transactions CASCADE;
TRUNCATE notify_logs CASCADE;
```

### 重置数据库

```bash
# 停止并删除容器和数据卷
docker-compose down -v

# 重新启动
docker-compose up -d

# 应用会自动运行迁移
```

### 手动运行迁移

```bash
# 进入容器
docker-compose exec postgres bash

# 执行迁移脚本
psql -U gopay -d gopay -f /path/to/migration.sql
```

### 备份数据库

```bash
# 备份到文件
docker-compose exec -T postgres pg_dump -U gopay gopay > backup_$(date +%Y%m%d_%H%M%S).sql

# 恢复备份
docker-compose exec -T postgres psql -U gopay gopay < backup_20240101_120000.sql
```

## 性能分析

### 查看慢查询

```sql
-- 启用慢查询日志
ALTER DATABASE gopay SET log_min_duration_statement = 1000;

-- 查看查询统计（需要 pg_stat_statements 扩展）
SELECT 
    query,
    calls,
    total_exec_time,
    mean_exec_time,
    max_exec_time
FROM pg_stat_statements
ORDER BY mean_exec_time DESC
LIMIT 10;
```

### 查看表大小

```sql
SELECT 
    tablename,
    pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename)) AS size
FROM pg_tables 
WHERE schemaname = 'public'
ORDER BY pg_total_relation_size(schemaname||'.'||tablename) DESC;
```

### 查看索引使用情况

```sql
SELECT 
    schemaname,
    tablename,
    indexname,
    idx_scan,
    idx_tup_read,
    idx_tup_fetch
FROM pg_stat_user_indexes
ORDER BY idx_scan DESC;
```

### 分析查询计划

```sql
EXPLAIN ANALYZE 
SELECT * FROM orders 
WHERE notify_status = 'pending' 
  AND retry_count < 5;
```

## 监控

### 查看活动连接

```sql
SELECT 
    pid,
    usename,
    application_name,
    client_addr,
    state,
    query,
    query_start
FROM pg_stat_activity
WHERE datname = 'gopay';
```

### 查看锁等待

```sql
SELECT 
    blocked_locks.pid AS blocked_pid,
    blocked_activity.usename AS blocked_user,
    blocking_locks.pid AS blocking_pid,
    blocking_activity.usename AS blocking_user,
    blocked_activity.query AS blocked_statement,
    blocking_activity.query AS blocking_statement
FROM pg_catalog.pg_locks blocked_locks
JOIN pg_catalog.pg_stat_activity blocked_activity ON blocked_activity.pid = blocked_locks.pid
JOIN pg_catalog.pg_locks blocking_locks 
    ON blocking_locks.locktype = blocked_locks.locktype
    AND blocking_locks.database IS NOT DISTINCT FROM blocked_locks.database
    AND blocking_locks.relation IS NOT DISTINCT FROM blocked_locks.relation
    AND blocking_locks.page IS NOT DISTINCT FROM blocked_locks.page
    AND blocking_locks.tuple IS NOT DISTINCT FROM blocked_locks.tuple
    AND blocking_locks.virtualxid IS NOT DISTINCT FROM blocked_locks.virtualxid
    AND blocking_locks.transactionid IS NOT DISTINCT FROM blocked_locks.transactionid
    AND blocking_locks.classid IS NOT DISTINCT FROM blocked_locks.classid
    AND blocking_locks.objid IS NOT DISTINCT FROM blocked_locks.objid
    AND blocking_locks.objsubid IS NOT DISTINCT FROM blocked_locks.objsubid
    AND blocking_locks.pid != blocked_locks.pid
JOIN pg_catalog.pg_stat_activity blocking_activity ON blocking_activity.pid = blocking_locks.pid
WHERE NOT blocked_locks.granted;
```

## 故障排查

### 连接失败

```bash
# 检查容器状态
docker-compose ps

# 查看容器日志
docker-compose logs postgres

# 测试连接
docker-compose exec postgres pg_isready -U gopay
```

### 迁移失败

```sql
-- 查看迁移历史
SELECT * FROM schema_migrations;

-- 手动标记迁移为已完成（谨慎操作！）
INSERT INTO schema_migrations (version) VALUES ('001_init_schema');
```

### 性能问题

```sql
-- 重建索引
REINDEX TABLE orders;

-- 更新统计信息
ANALYZE orders;

-- 清理死元组
VACUUM FULL orders;
```
